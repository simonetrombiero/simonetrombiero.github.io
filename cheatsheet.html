    +------------- oop basics --------------+
    |                                       |
    | class derived : public base {         |
    |   private:                            |
    |     std::string s;                    |
    |   public:                             |
    |     const std::string& get() const{   |
    |       return s;                       |
    |     }                                 |
    | };                                    |
    |                                       |
    |                                       |
    +---------------------------------------+
        

    +--------- virtual destructor ----------+
    |                                       |
    | class base  {                         |
    |   public:                             |
    |     ~base(){}                         |
    | };                                    |
    | classe derived : public base{         |
    |   public:                             |
    |     derived(){}                       |
    | };                                    |
    |                                       |
    |                                       |
    +---------------------------------------+
        

    +------- exception definition ----------+
    | class custom_except : std::exception{ |
    |   public:                             |
    |     custom_except(const string& m){   |
    |       message = m                     |
    |     }                                 |
    |     virtual const char* what() const{ |
    |       return message.c_str();         |
    |     }                                 |
    |   private:                            |
    |     string message;                   |
    | };                                    |
    +---------------------------------------+
        

    +-------- stl data structure -----------+
    |                                       |
    | std::vector // dynamic array          |
    | std::list // doubly-linked list       |
    | std::stack // stack                   |
    | std::queue // queue                   |
    | std::priority_queue // priority queue |
    | std::set // set                       |
    | std::map // map or dictionary         |
    | std::unordered_set // hash set        |
    | std::unordered_map // hash map        |
    |                                       |
    +---------------------------------------+
        

    +----------- smart pointers ------------+
    |                                       |
    | std::shared_ptr<int> ptr1 =           |
    |   std::make_shared<int>(5);           |
    | std::shared_ptr<int> ptr2 = ptr1;     |
    |                                       |
    | std::unique_ptr<int> ptr1 =           |
    |   std::make_unique<int>(5);           |
    | std::unique_ptr<int> ptr2 =           |
    |   std::move(ptr1);                    |
    |                                       |
    |                                       |
    +---------------------------------------+
        

    +---------- template class -------------+
    |                                       |
    |                                       |
    | template<typename T>                  |
    | class optional{                       |
    |   private:                            |
    |     T _value;                         |
    |   public:                             |
    |     T value() {return _value;}        |
    | };                                    |
    |                                       |
    |                                       |
    +---------------------------------------+
        

    +--------- lambda & functor ------------+
    |                                       |
    |                                       |
    | auto square =                         |
    |   [&](int x) { return x * x; };       |
    |                                       |
    | auto lambda = [this, x](){ ... };     |
    |                                       |
    | std::function<int()> add =            |
    |   [a]() { return a+2; };              |
    |                                       |
    |                                       |
    +---------------------------------------+
        

    +------------ concurrency --------------+
    |                                       |
    | // Using std::thread                  |
    | std::thread t1(printHello);           |
    | t1.join();                            |
    |                                       |
    | // Using std::async                   |
    | std::future<void> f =                 |
    |   std::async(std::launch::async,      |
    |     printHello);                      |
    | f.wait();                             |
    |                                       |
    +---------------------------------------+
        

    +-------------- iterators --------------+
    |                                       |
    | std::vector<int> myVector{ ... };     |
    |                                       |
    | // Iterator to traverse the vector    |
    | std::vector<int>::iterator it;        |
    | for (it = myVector.begin(); it        |
    |   != myVector.end(); ++it) {          |
    |   std::cout << *it << " ";            |
    | }                                     |
    |                                       |
    |                                       |
